Build and Deploy on Each Node

Node A (10.0.0.10) - Memory Buffer

Deploy:

# Build Consensus Alignment
nix build .#consensus-alignment

# Copy binary to /run/current-system/sw/bin/
cp result/bin/consensus-alignment /run/current-system/sw/bin/

# Deploy Hermes plugins (from submodule)
mkdir -p ~/.hermes/plugins/memory/
cp sidecars/hermes-agent-nous/plugins/memory/hermes-lcm/* ~/.hermes/plugins/memory/

Update Hermes Config:

# Create/update config
cat > ~/.hermes/config.yaml << 'EOF'
memory_providers:
  hermes-lcm:
    type: memory
    config:
      db_path: ~/.hermes/lcm.db
      max_context_tokens: 128000
      summary_interval: 32000
EOF

Restart Hermes:

hermes restart || hermes gateway restart

Node B (10.0.0.11) - Director (Current Node)

Deploy:

# Build Director's Forge
nix build .#directors-forge

# Build Goose Execution
nix build .#goose-execution

# Copy binaries
cp result/bin/directors-forge /run/current-system/sw/bin/
cp result/bin/goose-execution /run/current-system/sw/bin/

# Deploy Hermes plugins (VSB Router, Psy-core)
mkdir -p ~/.hermes/plugins/model-providers/
mkdir -p ~/.hermes/plugins/general/

cp sidecars/hermes-agent-nous/plugins/model-providers/sovereign-vsb/* ~/.hermes/plugins/model-providers/
cp sidecars/hermes-agent-nous/plugins/general/psy-core/* ~/.hermes/plugins/general/

Update Hermes Config:

cat > ~/.hermes/config.yaml << 'EOF'
model_providers:
  sovereign-vsb:
    type: model
    config:
      mesh_nodes:
        - id: node-a
          ip: 10.0.0.10
          port: 8000
          models: [falcon, embedding]
        - id: node-b
          ip: 10.0.0.11
          port: 9119
          models: [hermes-4, qwen3-vl]
        - id: node-c
          ip: 10.0.0.12
          port: 8080
          models: [voxcpm2, moshi]
        - id: node-d
          ip: 10.0.0.13
          port: 8000
          models: [qwen3.6, carnice, qwen2.5-coder-14b]
      tokenspeed_url: http://10.0.0.13:8000
      pulse_enabled: true

hooks:
  psy-core:
    type: general
    config:
      strict_mode: true
      audit_log_path: ~/.hermes/psy-audit.log

Restart Hermes:

hermes restart || hermes gateway restart
