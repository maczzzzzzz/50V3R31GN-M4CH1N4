3. UNIFIED VOICE PIPELINE ON NODE D (Already Built Earlier)

The existing voice-pipeline module now accepts input from both Omi and PC capture:

#[mcp_tool]
async fn process_voice(
    source: VoiceSource,   // "omi" or "pc-headset"
    audio: Vec<u8>,
    context: MemPalaceContext,
) -> Result<VoiceResult> {
    // 1. Hermes Whisper STT (base)
    let transcript = hermes_voice::transcribe(&audio).await?;

    // 2. VibeVoice enhancement
    let structured = vibevoice::enhance_asr(&transcript, &audio).await?;

    // 3. Hermes Agent processing
    let response = hermes::process(&structured.text).await?;

    // 4. VoxCPM2 TTS
    let audio_out = voxcpm::generate_tts(&response, voice_prompt = structured.style).await?;

    // 5. Log & audit
    let episode = DevEpisode::from_voice_interaction(&structured, &response, source);
    context.graph.store_episode(episode).await?;
    psy_core::audit_voice(&episode).await?;

    // 6. Send audio back to correct device
    match source {
        VoiceSource::Omi => omi_client::send_audio(&audio_out).await?,
        VoiceSource::PCHeadset => pc_client::send_audio(&audio_out).await?,
    }

    Ok(VoiceResult { transcript: structured.text, audio: audio_out })
}
